# Все значения берутся из переменных окружения (Railway Environment Variables)

OPENAI_API_KEY = os.environ["OPENAI_API_KEY"]
TELEGRAM_BOT_TOKEN = os.environ["TELEGRAM_BOT_TOKEN"]
TELEGRAM_CHAT_ID = os.environ["TELEGRAM_CHAT_ID"]
GOOGLE_DOC_ID = os.environ["GOOGLE_DOC_ID"]  # ID документа "Резюме встреч 2026"
ZOOM_VERIFICATION_TOKEN = os.environ.get("ZOOM_VERIFICATION_TOKEN", "")
ZOOM_CLIENT_ID = os.environ["ZOOM_CLIENT_ID"]
ZOOM_CLIENT_SECRET = os.environ["ZOOM_CLIENT_SECRET"]
ZOOM_ACCOUNT_ID = os.environ["ZOOM_ACCOUNT_ID"]

# Google Service Account credentials (JSON строка в переменной окружения)
GOOGLE_CREDENTIALS_JSON = os.environ["GOOGLE_CREDENTIALS_JSON"]

# Максимальный размер куска аудио для Whisper (24 МБ)
MAX_CHUNK_SIZE_MB = 24

# Промпт для генерации резюме
SUMMARY_PROMPT = """Ты — ассистент, который создаёт резюме рабочих встреч для компании Coin Post (криптомедиа).

На входе ты получаешь транскрипт встречи. На выходе — структурированное резюме в строго определённом формате.

ФОРМАТ РЕЗЮМЕ:

Резюме [тип встречи] с [имена участников]

Продолжительность: [X часов Y минут]

Обсудили:
1. [Первая тема/решение]
   - [Детали, подпункты, ответственные, дедлайны]
   - [Ещё детали]
2. [Вторая тема/решение]
   - [Детали]
3. ...

ПРАВИЛА:
- Определи имена участников из транскрипта. Если имя не удаётся определить, напиши "участник".
- Тип встречи определи из контекста: "планёрки", "собрания", "брейншторма", "созвона".
- Пиши конкретно: кто, что, когда. Не пиши абстрактно.
- Если упоминаются дедлайны, даты, ответственные — обязательно включи их.
- Если упоминаются конкретные инструменты, проекты, клиенты — включи названия.
- Пиши на русском языке.
- Не добавляй ничего, чего не было в транскрипте.
- Нумерация основных пунктов: 1. 2. 3. Подпункты через дефис.
- Не добавляй вступление или заключение, только структурированное резюме.

ТРАНСКРИПТ ВСТРЕЧИ:
{transcript}

ПРОДОЛЖИТЕЛЬНОСТЬ ВСТРЕЧИ: {duration} минут
"""

openai_client = OpenAI(api_key=OPENAI_API_KEY)


# ==================== ZOOM AUTH ====================

def get_zoom_access_token():
    """Получаем токен Zoom через Server-to-Server OAuth"""
    url = "https://zoom.us/oauth/token"
    response = requests.post(
        url,
        params={"grant_type": "account_credentials", "account_id": ZOOM_ACCOUNT_ID},
        auth=(ZOOM_CLIENT_ID, ZOOM_CLIENT_SECRET),
    )
    response.raise_for_status()
    return response.json()["access_token"]


# ==================== СКАЧИВАНИЕ АУДИО ====================

def download_recording(download_url, access_token):
    """Скачиваем аудио-файл записи Zoom во временный файл"""
    headers = {"Authorization": f"Bearer {access_token}"}
    response = requests.get(download_url, headers=headers, stream=True)
    response.raise_for_status()

    tmp = tempfile.NamedTemporaryFile(suffix=".mp4", delete=False)
    for chunk in response.iter_content(chunk_size=8192):
        tmp.write(chunk)
    tmp.close()
    return tmp.name


# ==================== НАРЕЗКА АУДИО ====================

def split_audio(file_path):
    """
    Нарезаем аудио на куски по ~24 МБ для Whisper API.
    Конвертируем в mp3 для уменьшения размера.
    """
    audio = AudioSegment.from_file(file_path)
    duration_ms = len(audio)

    # Экспортируем полный файл в mp3, чтобы оценить размер
    tmp_full = tempfile.NamedTemporaryFile(suffix=".mp3", delete=False)
    audio.export(tmp_full.name, format="mp3", bitrate="64k")
    file_size_mb = os.path.getsize(tmp_full.name) / (1024 * 1024)
    os.unlink(tmp_full.name)

    # Если файл меньше лимита, возвращаем как есть
    if file_size_mb <= MAX_CHUNK_SIZE_MB:
        tmp_single = tempfile.NamedTemporaryFile(suffix=".mp3", delete=False)
        audio.export(tmp_single.name, format="mp3", bitrate="64k")
        return [tmp_single.name]

    # Считаем сколько кусков нужно
    num_chunks = math.ceil(file_size_mb / MAX_CHUNK_SIZE_MB)
    chunk_duration_ms = duration_ms // num_chunks

    chunk_files = []
    for i in range(num_chunks):
        start = i * chunk_duration_ms
        end = min((i + 1) * chunk_duration_ms, duration_ms)
        chunk = audio[start:end]

        tmp_chunk = tempfile.NamedTemporaryFile(suffix=".mp3", delete=False)
        chunk.export(tmp_chunk.name, format="mp3", bitrate="64k")
        chunk_files.append(tmp_chunk.name)

    return chunk_files


# ==================== ТРАНСКРИПЦИЯ ====================

def transcribe_audio(file_paths):
    """Транскрибируем каждый кусок через Whisper API и склеиваем"""
    full_transcript = []

    for i, path in enumerate(file_paths):
        print(f"Транскрибируем кусок {i + 1}/{len(file_paths)}...")
        with open(path, "rb") as audio_file:
            response = openai_client.audio.transcriptions.create(
                model="whisper-1",
                file=audio_file,
                language="ru",
                prompt=(
                    "Coin Post, криптовалюта, блокчейн, DeFi, GameFi, "
                    "токенсейл, Telegram, YouTube, AMA, L1, L2, NFT, "
                    "биржа, кошелёк, стейкинг, airdrop"
                ),
            )
        full_transcript.append(response.text)
        # Удаляем временный файл
        os.unlink(path)

    return "\n\n".join(full_transcript)


# ==================== ГЕНЕРАЦИЯ РЕЗЮМЕ ====================

def generate_summary(transcript, duration_minutes):
    """Генерируем резюме через GPT-4o-mini"""
    prompt = SUMMARY_PROMPT.format(
        transcript=transcript, duration=duration_minutes
    )

    response = openai_client.chat.completions.create(
        model="gpt-4o-mini",
        messages=[{"role": "user", "content": prompt}],
        temperature=0.3,
        max_tokens=4000,
    )
    return response.choices[0].message.content


# ==================== GOOGLE DOCS ====================

def get_google_docs_service():
    """Подключаемся к Google Docs API через Service Account"""
    creds_dict = json.loads(GOOGLE_CREDENTIALS_JSON)
    creds = Credentials.from_service_account_info(
        creds_dict,
        scopes=["https://www.googleapis.com/auth/documents"],
    )
    return build("docs", "v1", credentials=creds)


def append_to_google_doc(summary_text, meeting_date):
    """
    Дописываем резюме в конец Google Doc.
    Добавляем разделитель и дату перед текстом резюме.
    """
    service = get_google_docs_service()

    # Получаем текущую длину документа
    doc = service.documents().get(documentId=GOOGLE_DOC_ID).execute()
    end_index = doc["body"]["content"][-1]["endIndex"] - 1

    # Формируем текст для вставки
    separator = "\n\n" + "=" * 60 + "\n"
    full_text = f"{separator}{meeting_date}\n\n{summary_text}\n"

    # Вставляем текст в конец документа
    requests_body = [
        {
            "insertText": {
                "location": {"index": end_index},
                "text": full_text,
            }
        }
    ]

    # Применяем форматирование даты (красный цвет, жирный)
    # Сначала вставляем текст, потом форматируем
    service.documents().batchUpdate(
        documentId=GOOGLE_DOC_ID,
        body={"requests": requests_body},
    ).execute()

    # Форматируем дату красным цветом
    date_start = end_index + len(separator)
    date_end = date_start + len(meeting_date)

    format_requests = [
        {
            "updateTextStyle": {
                "range": {"startIndex": date_start, "endIndex": date_end},
                "textStyle": {
                    "bold": True,
                    "foregroundColor": {
                        "color": {
                            "rgbColor": {"red": 1.0, "green": 0.0, "blue": 0.0}
                        }
                    },
                },
                "fields": "bold,foregroundColor",
            }
        },
        # Заголовок резюме — жирный
        {
            "updateTextStyle": {
                "range": {
                    "startIndex": date_end + 2,  # после \n\n
                    "endIndex": date_end + 2 + summary_text.index("\n"),
                },
                "textStyle": {"bold": True},
                "fields": "bold",
            }
        },
    ]

    service.documents().batchUpdate(
        documentId=GOOGLE_DOC_ID,
        body={"requests": format_requests},
    ).execute()

    print(f"Резюме добавлено в документ: {meeting_date}")


# ==================== TELEGRAM ====================

def send_telegram_notification(meeting_topic, meeting_date):
    """Отправляем уведомление в Telegram со ссылкой на документ"""
    doc_url = f"https://docs.google.com/document/d/{GOOGLE_DOC_ID}/edit"
    text = (
        f"✅ Резюме встречи готово!\n\n"
        f"📋 {meeting_topic}\n"
        f"📅 {meeting_date}\n\n"
        f"📄 {doc_url}"
    )

    url = f"https://api.telegram.org/bot{TELEGRAM_BOT_TOKEN}/sendMessage"
    requests.post(url, json={
        "chat_id": TELEGRAM_CHAT_ID,
        "text": text,
        "disable_web_page_preview": True,
    })


# ==================== ВЕБХУК ====================

@app.route("/webhook", methods=["POST"])
def zoom_webhook():
    """Обработчик вебхука от Zoom"""
    data = request.json

    # Zoom URL validation (при первой настройке вебхука)
    if data.get("event") == "endpoint.url_validation":
        import hashlib
        import hmac
        plain_token = data["payload"]["plainToken"]
        encrypted = hmac.new(
            ZOOM_VERIFICATION_TOKEN.encode(),
            plain_token.encode(),
            hashlib.sha256,
        ).hexdigest()
        return jsonify({"plainToken": plain_token, "encryptedToken": encrypted})

    # Обрабатываем только событие "запись готова"
    if data.get("event") != "recording.completed":
        return jsonify({"status": "ignored"}), 200

    payload = data["payload"]["object"]
    meeting_topic = payload.get("topic", "Встреча")
    duration = payload.get("duration", 0)  # в минутах

    # Ищем аудио-файл среди записей
    recording_files = payload.get("recording_files", [])
    audio_file = None
    for rf in recording_files:
        if rf.get("file_type") == "MP4" or rf.get("recording_type") == "audio_only":
            audio_file = rf
            break

    # Если нет отдельного аудио, берём первый доступный файл
    if not audio_file and recording_files:
        audio_file = recording_files[0]

    if not audio_file:
        print("Нет файлов для обработки")
        return jsonify({"status": "no files"}), 200

    download_url = audio_file["download_url"]

    # Дата встречи
    from datetime import datetime
    start_time = payload.get("start_time", "")
    if start_time:
        dt = datetime.fromisoformat(start_time.replace("Z", "+00:00"))
        meeting_date = dt.strftime("%d %B %Y").lstrip("0")
        # Русские названия месяцев
        months_ru = {
            "January": "января", "February": "февраля", "March": "марта",
            "April": "апреля", "May": "мая", "June": "июня",
            "July": "июля", "August": "августа", "September": "сентября",
            "October": "октября", "November": "ноября", "December": "декабря",
        }
        for eng, rus in months_ru.items():
            meeting_date = meeting_date.replace(eng, rus)
    else:
        meeting_date = datetime.now().strftime("%d.%m.%Y")

    try:
        # 1. Получаем токен Zoom
        print("Получаем токен Zoom...")
        access_token = get_zoom_access_token()

        # 2. Скачиваем аудио
        print("Скачиваем запись...")
        audio_path = download_recording(download_url, access_token)

        # 3. Нарезаем на куски
        print("Нарезаем аудио...")
        chunks = split_audio(audio_path)
        os.unlink(audio_path)  # удаляем оригинал
        print(f"Получилось {len(chunks)} кусков")

        # 4. Транскрибируем
        print("Транскрибируем...")
        transcript = transcribe_audio(chunks)
        print(f"Транскрипт готов: {len(transcript)} символов")

        # 5. Генерируем резюме
        print("Генерируем резюме...")
        summary = generate_summary(transcript, duration)
        print("Резюме готово")

        # 6. Дописываем в Google Doc
        print("Записываем в Google Doc...")
        append_to_google_doc(summary, meeting_date)

        # 7. Уведомляем в Telegram
        send_telegram_notification(meeting_topic, meeting_date)
        print("Готово!")

        return jsonify({"status": "success"}), 200

    except Exception as e:
        error_msg = f"❌ Ошибка обработки встречи: {str(e)}"
        print(error_msg)
        # Уведомляем об ошибке в Telegram
        url = f"https://api.telegram.org/bot{TELEGRAM_BOT_TOKEN}/sendMessage"
        requests.post(url, json={"chat_id": TELEGRAM_CHAT_ID, "text": error_msg})
        return jsonify({"status": "error", "message": str(e)}), 500
